227 template <class Compare>
228 NodeList Node::linearBy(Compare cmp) {
229  std::queue<Node *> nodes;
230  for (int i = 0, len = children.size(); i < len; i++) {
231  nodes.push(children[i]);
232  }
233 
234  NodeList linear;
235  linear.push_back(this);
236 
237  while (!nodes.empty()) {
238  linear.push_sorted(nodes.front(), cmp);
239 
240  // add children of the head node to the stack
241  NodeList ch = nodes.front()->getChildren();
242  for (int i = 0, len = ch.size(); i < len; i++) {
243  nodes.push(ch[i]);
244  }
245  nodes.pop();
246  }
247 
248  return linear;
249 }
