44 // The map_indexing_suite class is a predefined indexing_suite derived
45 // class for wrapping std::vector (and std::vector like) classes. It provides
46 // all the policies required by the indexing_suite (see indexing_suite).
47 // Example usage:
48 //
49 // class X {...};
50 //
51 // ...
52 //
53 // class_<std::map<std::string, X> >("XMap")
54 // .def(map_indexing_suite_v2<std::map<std::string, X> >())
55 // ;
56 //
57 // By default indexed elements are returned by proxy. This can be
58 // disabled by supplying *true* in the NoProxy template parameter.
59 //
60 template <class Container, bool NoProxy = false,
61  class DerivedPolicies =
62  detail::final_map_v2_derived_policies<Container, NoProxy> >
63 class map_indexing_suite_v2

77  // __getitem__ for std::pair
78  static object pair_getitem(value_type const& x, int i) {
79  if (i == 0 || i == -2) {
80  return object(x.first);
81  } else if (i == 1 || i == -1) {
82  return object(x.second);
83  } else {

85  throw_error_already_set();

134  // pop map[key], or throw an error if it doesn't exist
135  static object dict_pop(Container* x, index_type const& k) {
136  const_iterator it = x->find(k);
137  object result;
138  if (it != x->end()) {
139  result = object(it->second);
140  x->erase(it->first);
141  return result;
142  } else {
143  PyErr_SetString(PyExc_KeyError, "Key not found.");
144  throw_error_already_set();
145  return object(); // None
146  }
147  }

150  static object dict_pop_default(Container* x, index_type const& k,
151  object const& default_val) {

153  object result;
154  if (it != x->end()) {
155  result = object(it->second);

159  return default_val;

163  // pop a tuple, or throw an error if empty
164  static object dict_pop_item(Container* x) {
165  const_iterator it = x->begin();
166  object result;
167  if (it != x->end()) {
168  result = make_tuple(it->first, it->second);
169  x->erase(it->first);
170  return result;
171  } else {
172  PyErr_SetString(PyExc_KeyError, "No more items to pop");
173  throw_error_already_set();

178  // create a new map with given keys, initialialized to value
179  static object dict_fromkeys(object const& keys, object const& value) {
180  object newmap = object(Container());
181  int numkeys = extract<int>(keys.attr("__len__")());
182  for (int i = 0; i < numkeys; i++) { // 'cuz python is more fun in C++...
183  newmap.attr("__setitem__")(keys.attr("__getitem__")(i), value);
184  }
185  return newmap;
186  }

189  template <typename PyClassT>
190  struct init_factory {
191  typedef typename PyClassT::metadata::holder Holder;
192  typedef bp::objects::instance<Holder> instance_t;

194  // connect the PyObject to a wrapped C++ instance
195  // borrowed from boost/python/object/make_holder.hpp
196  static void make_holder(PyObject* p) {
197  void* memory =
198  Holder::allocate(p, offsetof(instance_t, storage), sizeof(Holder));
199 
200  try {
201  // this only works for blank () constructors
202  (new (memory) Holder(p))->install(p);
203  } catch (...) {
204  Holder::deallocate(p, memory);
205  throw;
206  }
207  }

222  // copy keys and values from dictlike object (anything with keys())
223  static void dict_update(object* x, object const& dictlike) {
224  object key;
225  object keys = dictlike.attr("keys")();
226  int numkeys = extract<int>(keys.attr("__len__")());
227  for (int i = 0; i < numkeys; i++) {
228  key = keys.attr("__getitem__")(i);
229  x->attr("__setitem__")(key, dictlike.attr("__getitem__")(key));
230  }
231  }

329  template <class Class>
330  static void extension_def(Class& cl) {
331  // Wrap the map's element (value_type)
332  std::string elem_name = "map_indexing_suite_v2_";
333  std::string cl_name;
334  object class_name(cl.attr("__name__"));
335  extract<std::string> class_name_extractor(class_name);
336  cl_name = class_name_extractor();
337  elem_name += cl_name;
338  elem_name += "_entry";
339 
340  typedef typename mpl::if_<
341  mpl::and_<mpl::or_<is_class<data_type>, is_class<data_type&> >,
342  mpl::bool_<!NoProxy> >,
343  return_internal_reference<>, default_call_policies>::type
344  get_data_return_policy;
345 
346  class_<value_type>(elem_name.c_str())
347  .def("__repr__", &DerivedPolicies::print_elem)
348  .def("data", &DerivedPolicies::get_data, get_data_return_policy(),
349  "K.data() -> the value associated with this pair.\n")
350  .def("key", &DerivedPolicies::get_key,
351  "K.key() -> the key associated with this pair.\n")
352  .def("__getitem__", &pair_getitem)
353  .def("__len__", &pair_len)
354  // if this is a data element, this won't work, but we assume that the
355  // keys will be standard hashable elements
356  .def("first", &DerivedPolicies::get_key,
357  "K.first() -> the first item in this pair.\n")
358  .def("second", &DerivedPolicies::get_data, get_data_return_policy(),
359  "K.second() -> the second item in this pair.\n");
360  // add convenience methods to the map
361 
362  cl
363  // declare constructors in descending order of arity
364  .def("__init__", init_factory<Class>::from_dict,
365  "Initialize with keys and values from a Python dictionary: "
366  "{'key':'value'}\n")
367  .def("__init__", init_factory<Class>::from_list,
368  "Initialize with keys and values as tuples in a Python list: "
369  "[('key','value')]\n")
370  .def(init<>()) // restore default constructor
371 
372  .def("keys", &keys, "D.keys() -> list of D's keys\n")
373  .def(
374  "has_key", &contains,
375  "D.has_key(k) -> True if D has a key k, else False\n") // don't
376  // re-invent
377  // the wheel
378  .def("values", &values, "D.values() -> list of D's values\n")
379  .def("items", &items,
380  "D.items() -> list of D's (key, value) pairs, as 2-tuples\n")
381  .def("clear", &Container::clear,
382  "D.clear() -> None. Remove all items from D.\n")
383  .def("copy", &copy, "D.copy() -> a shallow copy of D\n")
384  .def("get", dict_get,
385  dict_get_overloads(args("default_val"),
386  "D.get(k[,d]) -> D[k] if k in D, else d. d "
387  "defaults to None.\n"))
388  .def("pop", &dict_pop)
389  .def("pop", &dict_pop_default,
390  "D.pop(k[,d]) -> v, remove specified key and return the "
391  "corresponding value\nIf key is not found, d is returned if "
392  "given, otherwise KeyError is raised\n")
393  .def("popitem", &dict_pop_item,
394  "D.popitem() -> (k, v), remove and return some (key, value) pair "
395  "as a\n2-tuple; but raise KeyError if D is empty\n")
396  .def("fromkeys", &dict_fromkeys,
397  (cl_name + ".fromkeys(S,v) -> New " + cl_name +
398  " with keys from S and values equal to v.\n")
399  .c_str())
400  .staticmethod("fromkeys")
401  .def("update", &dict_update,
402  "D.update(E) -> None. Update D from E: for k in E: D[k] = E[k]\n")
403  .def("iteritems", make_transform<iteritems>(),
404  "D.iteritems() -> an iterator over the (key, value) items of D\n")
405  .def("iterkeys", make_transform<iterkeys>(),
406  "D.iterkeys() -> an iterator over the keys of D\n")
407  .def("itervalues", make_transform<itervalues>(),
408  "D.itervalues() -> an iterator over the values of D\n");
409  }
